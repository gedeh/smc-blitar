1. cara kopi dari awal
   - buat folder
   - git init
   - git config --global user.name "[nama kamu]"
   - git config --golbal user.email "[email kamu]"
   - git pull [url repo]

2. cara cek branch 
   - git branch
   cara tambah branch
   - git branch [nama branch kamu]
   cara pindah branch 
   - git checkout [nama branch kamu]

3. upload file
   - git add [nama file]
   - git commit -m "[catatan perubahan]"
   - git push blitar iwan

4. kalo error muncul waktu upload/push
   bikin alias
   - git remote add [nama alias repo] [url repo]
   bikin link upstream ke branch
   - git push --set-upstream [nama alias repo] [nama branch]
   download repo branch
   - git pull [nama alias repo] [nama branch]

5. setting git lewat proxy
   git config --global http.proxy http://<username>:<password>@<proxy-server-url>:<port>

6. unset proxy 
   git config --global --unset-all http.proxy # to remove it from the global config
   git config --unset-all http.proxy  # to remove it from the local repo config file
